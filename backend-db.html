class PlayTime(models.Model):
    objects = None
    start_time = models.DateTimeField(auto_now_add=True)
    end_time = models.DateTimeField(null=True, blank=True)
    duration = models.DateTimeField(null=True, blank=True)
    # Account 모델과의 일대다 관계 설정
    account = models.ForeignKey(Account, on_delete=models.CASCADE, related_name='playtimes')


    def __str__(self):
        return f'{self.start_time}, {self.end_time}, {self.duration}'
    class Meta:
        db_table = 'playtime'

def valid_email(email):
    # 이메일 정규 표현식 패턴
    pattern = re.compile('^.+@+.+\.+.+$')
    # 정규 표현식에 맞지 않으면 유효하지 않은 이메일
    if not pattern.match(email):
        return False, JsonResponse({'message': 'INVALID_EMAIL'}, status=400)
    return True, None

minimum_password_length = 8
# 패스워드 길이 검사 함수
def valid_password(password):
    # 패스워드 길이가 최소 길이보다 짧으면 유효하지 않음
    if len(password) < minimum_password_length:
        return False, JsonResponse({'message': 'SHORT_PASSWORD'}, status=400)
    return True, None

  1. POST 요청 처리: post 메서드는 클라이언트로부터 post 요청을 받아 처리합니다.
  1. 데이터 파싱: 클라이언트로부터 받은 JSON 데이터(이메일, 패스워드)를 파싱합니다. 이는 섭버가 이해할 수 있는 형태로 데이터를 변환하는 과정입니다.
  1. 데이터 검증: 입력된 이메일과 비밀번호가 제대로 제공되었는지 확인합니다. 만약 필수 데이터가 누락되었다면, 400 에러와 함께 오류 메세지를 반환합니다.
  1. 유효성 검사: 이메일과 비밀번호 형식이 유효한지 valid_email, vaild_password를 호출하여 검사합니다. 유효하지 않다면 오류 메세지와 함께 응답을 반환합니다.
  1. 중복 계정 검사: 이미 존재하는 이메일로 가입을 시도할 경우, DB에서 동일한 이메일을 가진 계정이 있는지 검사합니다. 이미 계정이 있다면 409에러와 오류 메세지를 반환합니다.
  1. 계정 생성: 모든 검증이 통과하면, 새로운 계정을 생성합니다. 이때 비밀번호는 bcrypt라이브러리를 사용하여 해시 처리하여 암호화해 저장합니다.
  1. 예외 처리: 코드에서는 KeyError 예외를 처리하고 있습니다. 이는 JSON 데이터에서 필요한 키(ex: email 또는 password)를 찾지 못했을 때 발생합니다. 
# 회원가입을 처리하는 클래스
class SignUpView(View):
    # POST 요청을 처리하는 메서드
    def post(self, request):
        # 클라이언트로부터 받은 JSON 데이터를 파싱
        data = json.loads(request.body)
        try:
            # 각 필드의 데이터를 가져옴
            email = data.get('email', None)
            password = data.get('password', None)
            # 필수 데이터가 없으면 400 에러 반환
            if not (password and email):
                return JsonResponse({'message': 'KEY_ERROR'}, status=400)

            # 유효성 검사
            valid, response = valid_email(email)
            if not valid:
                return response

            valid, response = valid_password(password)
            if not valid:
                return response

            # 이미 존재하는 계정인지 검사
            if Account.objects.filter(Q(email=email)).exists():
                return JsonResponse({'message': 'USER_ALREADY_EXISTS'}, status=409)

            # 새 계정 생성
            Account.objects.create(
                email=email,
                password=bcrypt.hashpw(password.encode('utf-8'), bcrypt.gensalt()).decode('utf-8')
            )
            return JsonResponse({'message': 'SUCCESS'}, status=200)

        except KeyError:
            # KeyError 예외 처리
            return JsonResponse({"message": "KEY_ERROR"}, status=400)

  1. 데이터 파싱: 클라이언트로부터 받은 JSON 데이터를 파싱합니다. 이 데이터는 사용자의 이메일과 패스워드를 포함합니다.
  1. 계정 찾기:데이터베이스에서 클라이언트로 부터 받은 이메일을 가진 계정을 찾습니다. 이메일을 기반으로 계정 검색을 진행합니다.

  1. 성공 응답: 사용자에게 JWT 토큰, 이메일, 성공 메세지를 반환합니다. 이는 사용자가 성공적으로 로그인했음을 나타냅니다.
  1. 비밀번호 불일치 처리: 입력된 패스워드가 데이터베이스에 저장된 패스워드와 일치하지 않는 경우, 401 에러와 함께 메세지를 반환합니다.
  1. 계정 미존재 처리: 제공된 계정을 찾지 못한 경우 401에러와 메세지를 반환합니다. 이는 사용자 계정이 존재하지 않음을 의미합니다.
  1. 예외 처리: JSON 데이터에서 필요한 키를 찾지 못하면 400에러와 함께 메세지를 반환합니다.
# 로그인을 처리하는 클래스
class LoginView(View):
    # POST 요청을 처리하는 메서드
    def post(self, request):
        # 클라이언트로부터 받은 JSON 데이터를 파싱
        data = json.loads(request.body)
        try:
            # 각 필드의 데이터를 가져옴
            email = data.get('email')
            password = data.get('password')

            # 이메일 또는 전화번호로 계정을 찾음
            if Account.objects.filter(Q(email=email)).exists():
                account = Account.objects.get(Q(email=email))

                # 비밀번호가 맞는지 검사
                if bcrypt.checkpw(password.encode('utf-8'), account.password.encode('utf-8')):
                    # JWT 토큰 생성
                    token = jwt.encode({'email': email}, my_settings.SECRET['secret'], algorithm='HS256')
                    # 토큰을 클라이언트에 반환
                    return JsonResponse({'message': 'SUCCESS', 'token': token, 'email': email}, status=200)

                # 비밀번호가 틀린 경우
                return JsonResponse({"message": "INVALID_PASSWORD"}, status=401)

            # 계정이 존재하지 않는 경우
            return JsonResponse({"message": "INVALID_USER"}, status=401)

        except KeyError:
            # KeyError 예외 처리
            return JsonResponse({"message": "KEY_ERROR"}, status=400)

class RankingDB(View):

            return JsonResponse({"message": str(e)}, status=500)

class RankingView(View):
    def post(self, request):
        data = json.loads(request.body)  # JSON 데이터 로드
        try:
            email = data.get('email')

            if not email:
                return JsonResponse({'message': 'KEY_ERROR'}, status=400)

            # 이메일을 사용하여 Account 모델에서 사용자 조회
            if Account.objects.filter(Q(email=email)).exists():
                account = Account.objects.get(Q(email=email))
                # 관련된 PlayTime 데이터 조회
                play_times = PlayTime.objects.filter(Q(account_id=account.id))

                durations = sorted([play_time.duration.strftime('%H:%M:%S') for play_time in play_times if play_time.duration is not None])[:5]
                return JsonResponse({'durations': durations, 'email': email}, status=200)
            else:
                return JsonResponse({'message': 'ACCOUNT_NOT_FOUND'}, status=404)

        except KeyError:
            return JsonResponse({"message": "KEY_ERROR"}, status=400)
        except ObjectDoesNotExist:
            return JsonResponse({"message": "ACCOUNT_NOT"}, status=403)
        except Exception as e:
            return JsonResponse({"message": str(e)}, status=500)

  1. 최단 플레이 시간 조회: PlayTime 모델을 사용하여 각 사용자별로 가장 짧은 게임 플레이 시간을 찾고, 관련된 Account 모델의 정보(사용자 이메일)와 함께 조회합니다.
  1. 상위 사용자 선정: 조회된 데이터 중에서 가장 짧은 시간을 기록한 상위 5명의 사용자를 선택합니다. 이때, 시간 데이터는 ‘시:분:초’ 형식으로 표시되도록 변환합니다.
  1. 결과 포맷팅: 선택된 상위 5명의 사용자 정보를 리스트 형태로 만들어, 각 사용자의 이메일과 플레이 시간을 포함시킵니다.
class Ranking_all_View(View): 
    # 뷰 클래스 선언 (장고의 View 클래스를 상속받음)

    def post(self, request): 
        # POST 요청을 처리하는 메소드
        try: 
            # 예외 처리를 위한 try 블록 시작

            # PlayTime 모델에서 각 유저별로 가장 짧은 duration을 찾고, 관련된 Account 정보를 함께 조회
            shortest_durations_per_user = PlayTime.objects.select_related('account').annotate(
                shortest_duration=Min('duration')
            ).values('account__email', 'shortest_duration')

            # 전체에서 가장 짧은 상위 5개의 durations을 가진 유저 선택
            top_users = (
                shortest_durations_per_user
                .order_by('shortest_duration')[:5]
                .annotate(duration_str=Trunc('shortest_duration', 'second'))
            )

            # 결과를 리스트 형태로 만들어서 각 사용자의 이메일과 플레이 시간을 포함
            result = [
                {
                    'email': user['account__email'],
                    'duration': user['duration_str'].strftime('%H:%M:%S')
                } for user in top_users
            ]

            # JsonResponse를 사용하여 결과를 JSON 형태로 반환, 상태 코드 200
            return JsonResponse({'top_users': result}, status=200)

        except Exception as e:
            # 예외 발생시 JsonResponse로 예외 메시지 반환, 상태 코드 500
            return JsonResponse({"message": str(e)}, status=500)
